MyBatis(二)：使用mapper代理方法进行增删改查操作

发表于 2017-06-14 | 分类于 JavaWeb

一、原始的DAO开发方法

原始的DAO开发方法需要程序员自行开发DAO接口和DAO的实现类，存在不少弊端 :

  1. DAO接口实现类方法中存在大量模板方法，设想能否将这些代码提取出来，大大减轻程序员的工作量。
  2. 调用sqlsession方法时将statement的id硬编码了,不利于系统维护
  3. 调用sqlsession方法时传入的变量，由于sqlsession方法使用泛型，即使变量类型传入错误，在编译阶段也不报错，不利于程序员开发。

二、使用mapper代理的方法开发DAO

mapper代理方法思路：
程序员只需开发mapper接口(相对于DAO接口)，而具体的实现类则由MyBatis通过与mapper接口对应的映射文件(mapper.xml)来生成代理对象，则可以通过这个代理对象来与数据库实现交互。

2.1 mapper代理方法需要遵循的规范

  1. mapper.xml中的命名空间(namespace)需要和mapper接口的地址一致
  2. mapper.xml中的statement(sql语句)的id需要和mapper接口中对应的方法名一致
  3. mapper.xml中的statement的parameterType和mapper接口中对应方法的输入参数类型一致
  4. mapper.xml中的statement的resultType和mapper接口中对应方法的返回值类型一致(若接口方法返回的是List类型，则resultType应是List中元素的类型)

2.2.2 全局配置文件 mybatis-config.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
  PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
  "http://mybatis.org/dtd/mybatis-3-config.dtd">


<configuration>
    <!-- 外部属性文件 -->
    <properties resource="config/mysql.properties"></properties>
    <!-- 数据源及事务管理 -->
    <environments default="development">
        <environment id="development">
            <transactionManager type="JDBC" />
            <dataSource type="POOLED">
                <property name="driver" value="${driver}" />
                <property name="url" value="${url}" />
                <property name="username" value="${username}" />
                <property name="password" value="${password}" />
            </dataSource>
        </environment>
    </environments>
    <!-- 映射文件注册 -->
    <mappers>
        <mapper resource="config/mapper/UserMapper.xml"/>
    </mappers>
</configuration>

2.2.3 UserMapper接口

package com.jzf.mybatis.mapper;

import com.jzf.mybatis.domain.User;

public interface UserMapper {

    public User findUserById(int id);

    public void insertUser(User user);

    public void updateUser(User user);

    public void deleteUserById(int id);
}

2.2.4 UserMapper.xml映射文件

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper
  PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
  "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<!-- namespace -->
<mapper namespace="com.jzf.mybatis.mapper.UserMapper">
    <!-- 
        select语句 id:唯一表示一条sql语句 
        parameterType:输入参数类型
        resultType:输出类型
        #{}:输入参数占位符 
    -->
    <select id="findUserById" parameterType="int" resultType="com.jzf.mybatis.domain.User">
        select * from user where id=#{id}
    </select>
    <!--
         插入语句
        useGeneratedKeys:仅对 insert和 update有用,
        设置为true,会令 MyBatis使用 JDBC的 getGeneratedKeys 方法来取出由数据库内部生成的主键
    -->
    <insert id="insertUser" parameterType="com.jzf.mybatis.domain.User" useGeneratedKeys="true" keyProperty="id">
        insert into user(name,password) value(#{name},#{password})
    </insert>
    <!--
         更新语句
    -->
    <update id="updateUser" parameterType="com.jzf.mybatis.domain.User">
        update user set name=#{name},password=#{password} where id=#{id}
    </update>
    <!--
         删除语句
    -->
    <delete id="deleteUserById" parameterType="int">
        delete from user where id=#{id}
    </delete>    
</mapper>

2.2.5 测试程序

package com.jzf.mybatis.test;

import static org.junit.Assert.*;

import java.io.InputStream;

import org.apache.ibatis.io.Resources;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import org.junit.Before;
import org.junit.Test;

import com.jzf.mybatis.domain.User;
import com.jzf.mybatis.mapper.UserMapper;

public class UserMapperTest2 {
    private SqlSessionFactory sqlSessionFactory;

    @Before
    public void setUp() throws Exception {
        String resource = "config/mybatis-config.xml";
        InputStream inputStream = Resources.getResourceAsStream(resource);
        // 读取配置文件并创建会话工厂
        sqlSessionFactory = new SqlSessionFactoryBuilder().build(inputStream);

    }

    @Test
    public void testFindUserById() {
        SqlSession sqlSession = null;
        User user = null;
        try {
            sqlSession = sqlSessionFactory.openSession();
            // 获得MyBatis自动生成的代理对象
            UserMapper userMapper = sqlSession.getMapper(UserMapper.class);
            // 通过代理对象执行响应的方法
            user = userMapper.findUserById(1);
            System.out.println(user);
        } finally {
            sqlSession.close();
        }

    }

    @Test
    public void testInsertUser() {
        SqlSession sqlSession = null;
        User user = new User("王二麻子", "333333");
        try {
            sqlSession = sqlSessionFactory.openSession();
            // 获得MyBatis自动生成的代理对象
            UserMapper userMapper = sqlSession.getMapper(UserMapper.class);
            // 通过代理对象执行响应的方法
            userMapper.insertUser(user);
            // 执行除了查询以外的操作必须提交
            sqlSession.commit();
            //查看是否已获得自增主键
            System.out.println(user);
        } finally {
            sqlSession.close();
        }
    }

    @Test
    public void testUpdateUser() {
        SqlSession sqlSession = null;
        User user = null;
        try {
            sqlSession = sqlSessionFactory.openSession();
            // 获得MyBatis自动生成的代理对象
            UserMapper userMapper = sqlSession.getMapper(UserMapper.class);
            //获取用户
            user = userMapper.findUserById(1);
            //更新用户
            user.setName("小张三");
            userMapper.updateUser(user);
            sqlSession.commit();
            //再次获取用户
            user = userMapper.findUserById(1);
            System.out.println(user);
        } finally {
            sqlSession.close();
        }
    }

    @Test
    public void testDeleteUser() {
        SqlSession sqlSession = null;
        try {
            sqlSession = sqlSessionFactory.openSession();
            // 获得MyBatis自动生成的代理对象
            UserMapper userMapper = sqlSession.getMapper(UserMapper.class);
            // 通过代理对象执行响应的方法
            userMapper.deleteUserById(9);
            sqlSession.commit();
        } finally {
            sqlSession.close();
        }
    }
}
